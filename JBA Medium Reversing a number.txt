//Write a program that reads a three-digit number, 
//reverses the order of its digits, and outputs a new num


fun main() {
    
    val base = readln().toInt()
    val num1 = base % 10 // result 9
    val num2 = (base % 100)/10 //result 2
    val num3 = base / 100 //result 6
    val result = "$num1$num2$num3"
    println(result)
}
